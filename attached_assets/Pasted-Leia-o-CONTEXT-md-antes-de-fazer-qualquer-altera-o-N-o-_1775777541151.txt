### Disparar emails nos eventos

1. POST /auth/register → após criar negócio com sucesso:
```typescript
import { sendEmail, emails } from '../services/email';
const tpl = emails.boasVindas(body.name, body.businessName);
await sendEmail(body.email, tpl.subject, tpl.html);
```

2. PATCH /admin/businesses/:id (aprovação) → quando status muda para 'active':
```typescript
if (body.status === 'active' && business.ownerEmail) {
  const tpl = emails.cadastroAprovado(business.ownerName || 'Lojista', business.name);
  await sendEmail(business.ownerEmail, tpl.subject, tpl.html);
}
if (body.status === 'rejected' && business.ownerEmail) {
  const tpl = emails.cadastroRejeitado(business.ownerName || 'Lojista', business.name, body.rejectionReason || 'Não informado');
  await sendEmail(business.ownerEmail, tpl.subject, tpl.html);
}
```

3. Webhook Stripe checkout.session.completed → pagamento confirmado:
```typescript
const planLabel = plan === 'base' ? 'Base' : 'Premium';
const valor = plan === 'base' ? 'R$59,90' : 'R$89,90';
const tpl = emails.pagamentoConfirmado(business.ownerName || 'Lojista', planLabel, valor);
await sendEmail(business.ownerEmail, tpl.subject, tpl.html);
```

4. Webhook Stripe invoice.payment_failed → após downgrade:
```typescript
const tpl = emails.pagamentoFalhou(business.ownerName || 'Lojista', sub.plan);
await sendEmail(business.ownerEmail, tpl.subject, tpl.html);
```

5. POST /businesses/:id/review → após salvar avaliação:

## PARTE 2 — ESQUECI MINHA SENHA

### Schema — campo novo em business_users
Adicionar via Drizzle migration:
- passwordResetToken: text nullable
- passwordResetExpiresAt: timestamp nullable

### Backend — 2 rotas novas

POST /api/auth/forgot-password
Body: { email: string }
Rate limit: 3 por hora por IP (usar registerLimiter ou criar novo)

```typescript
// 1. Buscar business_user pelo email
const user = await db.query.businessUsers.findFirst({
  where: eq(businessUsers.email, body.email)
});

// Sempre retornar 200 — não revelar se email existe
if (!user) {
  return res.json({ message: 'Se o email existir, você receberá as instruções.' });
}

// 2. Gerar token único
const token = crypto.randomBytes(32).toString('hex');
const expiresAt = new Date(Date.now() + 60 * 60 * 1000); // 1 hora

// 3. Salvar token no banco
await db.update(businessUsers)
  .set({ passwordResetToken: token, passwordResetExpiresAt: expiresAt })
  .where(eq(businessUsers.id, user.id));

// 4. Buscar dados do negócio para o email
const business = await db.query.businesses.findFirst({
  where: eq(businesses.id, user.businessId)
});

// 5. Enviar email
const tpl = emails.recuperacaoSenha(business?.ownerName || 'Lojista', token);
await sendEmail(body.email, tpl.subject, tpl.html);

return res.json({ message: 'Se o email existir, você receberá as instruções.' });
```

POST /api/auth/reset-password
Body: { token: string, newPassword: string }

```typescript
// 1. Buscar user pelo token
const user = await db.query.businessUsers.findFirst({
  where: eq(businessUsers.passwordResetToken, body.token)
});

if (!user || !user.passwordResetExpiresAt || user.passwordResetExpiresAt < new Date()) {
  return res.status(400).json({ error: 'Token inválido ou expirado.' });
}

if (body.newPassword.length < 8) {
  return res.status(400).json({ error: 'Senha deve ter mínimo 8 caracteres.' });
}

// 2. Atualizar senha e limpar token
const hash = await bcrypt.hash(body.newPassword, 10);
await db.update(businessUsers)
  .set({
    passwordHash: hash,
    passwordResetToken: null,
    passwordResetExpiresAt: null
  })
  .where(eq(businessUsers.id, user.id));

return res.json({ message: 'Senha atualizada com sucesso.' });
```

## PARTE 3 — PÁGINA ADMIN DE ASSINATURAS

### Backend — GET /api/admin/subscriptions

```typescript
// Retorna visão completa de assinaturas
const subs = await db.query.subscriptions.findMany({
  with: { business: true }
});

// Calcular MRR
const planPrices = { base: 59.90, premium: 89.90 };
const activeSubs = subs.filter(s => s.status === 'active');
const mrr = activeSubs.reduce((sum, s) => sum + (planPrices[s.plan] || 0), 0);

// Agrupar por status
const byStatus = {
  active: subs.filter(s => s.status === 'active').length,
  past_due: subs.filter(s => s.status === 'past_due').length,
  cancelled: subs.filter(s => s.status === 'cancelled').length,
};

return res.json({ mrr, byStatus, subscriptions: subs });
```
