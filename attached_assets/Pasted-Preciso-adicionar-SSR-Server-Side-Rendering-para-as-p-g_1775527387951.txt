A rota /negocio/:id precisa retornar HTML completo com os dados do negócio

- Banco: PostgreSQL + Drizzle

### 2. Criar o entry server no frontend
Criar arquivo artifacts/hub-londrina/src/entry-server.tsx:

import React from 'react'
import { renderToString } from 'react-dom/server'
import { Router } from 'wouter'
import staticLocationHook from 'wouter/static-location'
import App from './App'

export function render(url: string, empresaData: any) {
  const html = renderToString(
    <Router hook={staticLocationHook(url)}>
      <App prefetchedData={empresaData} />
    </Router>
  )
  return html
}

### 5. Adicionar rota SSR no Express
Em artifacts/api-server/src/index.ts (ou routes/), adicionar ANTES 
das rotas de API:

import { createServer as createViteServer } from 'vite'
import fs from 'fs'
import path from 'path'

// Em desenvolvimento: usa Vite como middleware
// Em produção: serve o bundle SSR compilado

app.get('/negocio/:id', async (req, res) => {
  try {
    const id = Number(req.params.id)
    
    // Busca os dados do negócio (mesma lógica da API GET /api/businesses/:id)
    const business = await getBusinessById(id) // função já existente
    
    if (!business) {
      return res.status(404).send('Negócio não encontrado')
    }

    // Renderiza o HTML com os dados
    let render
    if (process.env.NODE_ENV === 'production') {
      // Carrega o bundle SSR compilado
      const module = await import('../../../hub-londrina/dist/server/entry-server.js')
      render = module.render
    } else {
      // Em dev, usa Vite SSR transform
      const vite = await createViteServer({ server: { middlewareMode: true } })
      const module = await vite.ssrLoadModule('/src/entry-server.tsx')
      render = module.render
    }

    const appHtml = render(`/negocio/${id}`, business)
    
    // Lê o template HTML
    const template = fs.readFileSync(
      path.resolve('../hub-londrina/index.html'), 'utf-8'
    )
    
    // Injeta o HTML renderizado e os dados para hidratação
    const html = template
      .replace('<!--ssr-outlet-->', appHtml)
      .replace(
        '</head>',
        `<script>window.__SSR_DATA__ = ${JSON.stringify(business)}</script>
         <meta name="description" content="${business.description || business.name + ' em Londrina'}">
         <title>${business.name} — Hub Londrina</title>
         </head>`
      )
    
    res.status(200).set({ 'Content-Type': 'text/html' }).end(html)
    
  } catch (e) {
    console.error(e)
    res.status(500).send('Erro interno')
  }
})

### 7. Hidratação no cliente
Em artifacts/hub-londrina/src/main.tsx, adaptar para hidratar 
quando dados SSR estiverem presentes:

import React from 'react'
import { hydrateRoot, createRoot } from 'react-dom/client'
import App from './App'

const ssrData = (window as any).__SSR_DATA__
const container = document.getElementById('root')!

if (ssrData) {
  // Hidrata o HTML já renderizado pelo servidor
  hydrateRoot(container, <App prefetchedData={ssrData} />)
} else {
  // Renderização normal SPA
  createRoot(container).render(<App />)
}
